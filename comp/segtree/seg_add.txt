template<typename T>
class Add : virtual public LazySegmentTree<T> {
protected:
	void update_one(Mono<T>& pos, T& x) {
		pos.val += x;
		pos.lazy += x;
	}
	void fold(Mono<T>& pos) { pos.lazy = 0; }
	void propagate(Mono<T>& par, Mono<T>& child) {
		child.val += par.lazy;
		child.lazy += par.lazy;
	}
};
